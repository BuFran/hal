209 INLINE void pin_clock_enable(const uint32_t pin)
210 {
211  switch (_pin_port(pin)) {
212 #if defined(GPIO_PORT_A_BASE)
213  case GPIOA: rcc_periph_clock_enable(RCC_GPIOA); break;
214 #endif
215 #if defined(GPIO_PORT_B_BASE)
216  case GPIOB: rcc_periph_clock_enable(RCC_GPIOB); break;
217 #endif
218 #if defined(GPIO_PORT_C_BASE)
219  case GPIOC: rcc_periph_clock_enable(RCC_GPIOC); break;
220 #endif
221 #if defined(GPIO_PORT_D_BASE)
222  case GPIOD: rcc_periph_clock_enable(RCC_GPIOD); break;
223 #endif
224 #if defined(GPIO_PORT_E_BASE)
225  case GPIOE: rcc_periph_clock_enable(RCC_GPIOE); break;
226 #endif
227 #if defined(GPIO_PORT_F_BASE)
228  case GPIOF: rcc_periph_clock_enable(RCC_GPIOF); break;
229 #endif
230 #if defined(GPIO_PORT_G_BASE)
231  case GPIOG: rcc_periph_clock_enable(RCC_GPIOG); break;
232 #endif
233 #if defined(GPIO_PORT_H_BASE)
234  case GPIOH: rcc_periph_clock_enable(RCC_GPIOH); break;
235 #endif
236 #if defined(GPIO_PORT_I_BASE)
237  case GPIOI: rcc_periph_clock_enable(RCC_GPIOI); break;
238 #endif
239  default:
240  break;
241  }
242 }
